Required:
-command=img2pdf
    Must be exactly "img2pdf"

-input="folder_path"
    Path to folder containing images

-output="output_file.pdf"
    Output PDF file path


Optional:

-margin=NUMBER
    Page margin in millimeters
    Default: 8

-numberOffset=NUMBER
    Distance from bottom-left corner for page number (mm)
    Default: 4

-stretch=y
    Force image to fill available area (distort allowed)
    Default: disabled

-percentThreshold=NUMBER
    Used only with -pageSize=auto
    If image size difference from ISO paper size is within this percent,
    snap page size to that ISO size
    Default: 5

-pageSize=VALUE
    Supported values:

    1_1
        Page size = image size + 2×margin
        Stretch disabled automatically

    auto
        Try to match ISO paper size (A5–2A0) within percentThreshold

    A5
    A4
    A3
    A2
    A1
    A0
    2A0
        Force fixed ISO paper size

-fitWidth=NUMBER
    Force page width (in mm)
    Page height calculated proportionally
    Overrides pageSize and auto
